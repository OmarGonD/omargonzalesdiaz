useState - Class based

Solo se debe usar useState en componentes basados en clases.

import React, { Component } from 'react'

class App extends Component {

  state = {
    persons: [
      {name: 'Max', age:28 },
      {name: 'Manu', age:29 },
      {name: 'Stephanie', age:26 }
    ],
    otherstate: 'some other value'
  }

};
